732  static inline void toOption(Modifier const& mod, bool x,
733  Uint const * J_V, char const * J_N, SIZE_T SJ,
734  Uint const * V, char const * N, SIZE_T S,
735  Uint* po, Uint* jo,
736  int* en, SIZE_T* eo
737  ){
738  //loop through mod
739  SIZE_T n = mod.length();
740  for (SIZE_T i = 0; i < n; ++i) {
741  //First check for JPCRE2 mods
742  for(SIZE_T j = 0; j < SJ; ++j){
743  if(J_N[j] == mod[i]) {
744  if(x) *jo |= J_V[j];
745  else *jo &= ~J_V[j];
746  goto endfor;
747  }
748  }
749 
750  //Now check for PCRE2 mods
751  for(SIZE_T j = 0; j< S; ++j){
752  if(N[j] == mod[i]){
753  if(x) *po |= V[j];
754  else *po &= ~V[j];
755  goto endfor;
756  }
757  }
758 
759  //Modifier didn't match, invalid modifier
760  *en = (int)ERROR::INVALID_MODIFIER;
761  *eo = (int)mod[i];
762 
763  endfor:;
764  }
765  }

790  Uint po, Uint jo

804  }
